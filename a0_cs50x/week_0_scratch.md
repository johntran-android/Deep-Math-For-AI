> 100 = 4***1** + 2***0** + 1***0** = 4 + 0 + 0 = 4
> 101 = 4***1** + 2***0** + 1***1** = 4 + 0 + 1 = 5
> 110 = 4***1** + 2***1** + 1***0** = 4 + 2 + 0 = 6
> 111 = 4***1** + 2***1** + 1***1** = 4 + 2 + 1 = 7
